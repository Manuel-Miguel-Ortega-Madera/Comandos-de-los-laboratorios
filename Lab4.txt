sudo nano /etc/apache2/sites-available/hello.conf
a2ensite hello.conf
sudo systemctl reload apache2
sudo mkdir /var/www/credencial
sudo nano /var/www/credencial/index.html <h1>[contenido]<h1>

sudo nano /etc/apache2/sites-available/credencial.conf
a2ensite credencial.conf
sudo systemctl reload o restart apache2

sudo apt update -y && sudo apt upgrade -y
sudo apt install mailutils -y
sudo apt install postfix -y
cd /etc/postfix/sasl
sudo nano /etc/postfix/sasl/sasl_passwd
sudo postmap /etc/postfix/sasl/sasl_passwd
sudo chmod 600 /etc/postfix/sasl/sasl_passwd
sudo chmod 600 /etc/postfix/sasl/sasl_passwd.db
sudo nano /etc/postfix/main.cf
sudo systemctl restart postfix
sudo systemctl status postfix
